# ───────────────────────────────────────────────────────────────────────────────
#  TABLE: Sessions
#  Purpose: Stores active user sessions with HttpOnly cookie authentication
# ───────────────────────────────────────────────────────────────────────────────

TABLE_NAME: Sessions

COLUMNS:
┌─────────────────────┬──────────────┬──────────┬─────────────────────────────────────────┐
│ Column Name         │ Data Type    │ Required │ Description                             │
├─────────────────────┼──────────────┼──────────┼─────────────────────────────────────────┤
│ ROWID               │ bigint       │ Auto     │ Primary key (auto-generated by Catalyst)│
│ Session_ID          │ text         │ Yes      │ Unique session identifier (43 chars)    │
│ User_ID             │ text         │ Yes      │ Reference to Users.ROWID                │
│ User_Email          │ text         │ Yes      │ User's email (denormalized for queries) │
│ User_Role           │ text         │ Yes      │ User's role (User/Admin)                │
│ IP_Address          │ text         │ No       │ Client IP address                       │
│ User_Agent          │ text         │ No       │ Browser/device User-Agent (max 500)     │
│ Device_Fingerprint  │ text         │ No       │ SHA-256 hash of device fingerprint      │
│ CSRF_Token          │ text         │ Yes      │ CSRF token for this session (43 chars)  │
│ Created_At          │ text         │ Yes      │ ISO 8601 timestamp of session creation  │
│ Last_Accessed_At    │ text         │ Yes      │ ISO 8601 timestamp of last activity     │
│ Expires_At          │ text         │ Yes      │ ISO 8601 timestamp of session expiry    │
│ Is_Active           │ boolean      │ Yes      │ true = active, false = revoked          │
└─────────────────────┴──────────────┴──────────┴─────────────────────────────────────────┘

INDEXES (Create in Catalyst Console):
  - Session_ID (Unique) - Primary lookup for session validation
  - User_ID             - Find all sessions for a user
  - Is_Active           - Filter active sessions
  - Expires_At          - Cleanup expired sessions

# ───────────────────────────────────────────────────────────────────────────────
#  TABLE: Session_Audit_Log
#  Purpose: Security audit trail for session-related events
# ───────────────────────────────────────────────────────────────────────────────

TABLE_NAME: Session_Audit_Log

COLUMNS:
┌─────────────────────┬──────────────┬──────────┬─────────────────────────────────────────┐
│ Column Name         │ Data Type    │ Required │ Description                             │
├─────────────────────┼──────────────┼──────────┼─────────────────────────────────────────┤
│ ROWID               │ bigint       │ Auto     │ Primary key (auto-generated by Catalyst)│
│ Event_Type          │ text         │ Yes      │ Type of event (see list below)          │
│ Session_ID          │ text         │ No       │ Related session ID (may be partial)     │
│ User_ID             │ text         │ No       │ Related user's ROWID                    │
│ User_Email          │ text         │ No       │ User's email at time of event           │
│ IP_Address          │ text         │ No       │ Client IP address                       │
│ Event_Timestamp     │ text         │ Yes      │ ISO 8601 timestamp of event             │
│ Details             │ text         │ No       │ JSON string with additional event data  │
│ Severity            │ text         │ No       │ INFO, WARNING, ERROR, CRITICAL          │
└─────────────────────┴──────────────┴──────────┴─────────────────────────────────────────┘

EVENT_TYPES:
  - SESSION_CREATED       : New session created on login/register
  - SESSION_VALIDATED     : Session successfully validated
  - SESSION_EXPIRED       : Session expired (timeout)
  - SESSION_REVOKED       : Session manually revoked by user
  - SESSION_REVOKED_ADMIN : Session revoked by administrator
  - SESSION_LIMIT_ENFORCED: Oldest session revoked due to concurrent limit
  - CSRF_VALIDATION_FAILED: CSRF token mismatch detected
  - SESSION_INVALID       : Invalid session ID presented
  - PASSWORD_CHANGED      : User changed password (all sessions revoked)
  - SUSPICIOUS_ACTIVITY   : Potential security threat detected

INDEXES (Create in Catalyst Console):
  - User_ID             - Find all events for a user
  - Event_Type          - Filter by event type
  - Event_Timestamp     - Time-based queries

EXAMPLE RECORD:
{
    "ROWID": "111222333444555",
    "Event_Type": "SESSION_CREATED",
    "Session_ID": "dGhpc2lz...xyz",
    "User_ID": "987654321098765",
    "User_Email": "user@example.com",
    "IP_Address": "192.168.1.100",
    "Event_Timestamp": "2026-03-31T12:00:00.000Z",
    "Details": "{\"user_agent\": \"Chrome/120.0\"}",
    "Severity": "INFO"
}
