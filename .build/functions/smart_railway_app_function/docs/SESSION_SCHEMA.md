# ───────────────────────────────────────────────────────────────────────────────
#  TABLE: Sessions
#  Purpose: Stores active sessions for BOTH passengers AND employees
#  
#  AUTHENTICATION FLOW:
#  - Passengers: Login via /session/login → creates session with User_Type='user'
#  - Employees:  Login via /session/employee/login → creates session with User_Type='employee'
# ───────────────────────────────────────────────────────────────────────────────

TABLE_NAME: Sessions

COLUMNS:
┌─────────────────────┬──────────────┬──────────┬───────────────────────────────────────────────────┐
│ Column Name         │ Data Type    │ Required │ Description                                       │
├─────────────────────┼──────────────┼──────────┼───────────────────────────────────────────────────┤
│ ROWID               │ bigint       │ Auto     │ Primary key (auto-generated by Catalyst)          │
│ Session_ID          │ text         │ Yes      │ Unique session identifier (43 chars URL-safe)     │
│ User_ID             │ text         │ Yes      │ Reference to Users.ROWID OR Employees.ROWID       │
│ User_Type           │ text         │ Yes      │ 'user' (passenger) OR 'employee' (staff/admin)    │
│ User_Email          │ text         │ Yes      │ User's email (denormalized for queries)           │
│ User_Role           │ text         │ Yes      │ 'Passenger' OR 'Employee' OR 'Admin'              │
│ IP_Address          │ text         │ No       │ Client IP address                                 │
│ User_Agent          │ text         │ No       │ Browser/device User-Agent (max 500)               │
│ Device_Fingerprint  │ text         │ No       │ SHA-256 hash of device fingerprint                │
│ CSRF_Token          │ text         │ Yes      │ CSRF token for this session (43 chars)            │
│ Created_At          │ text         │ Yes      │ ISO 8601 timestamp of session creation            │
│ Last_Accessed_At    │ text         │ Yes      │ ISO 8601 timestamp of last activity               │
│ Expires_At          │ text         │ Yes      │ ISO 8601 timestamp of session expiry              │
│ Is_Active           │ boolean      │ Yes      │ true = active, false = revoked                    │
└─────────────────────┴──────────────┴──────────┴───────────────────────────────────────────────────┘

USER_TYPE VALUES:
  - 'user'     : Session belongs to a PASSENGER (references Users table)
  - 'employee' : Session belongs to STAFF/ADMIN (references Employees table)

USER_ROLE VALUES:
  - 'Passenger' : Regular passenger (User_Type='user')
  - 'Employee'  : Staff member (User_Type='employee')
  - 'Admin'     : Administrator (User_Type='employee')

INDEXES (Create in Catalyst Console):
  - Session_ID (Unique) - Primary lookup for session validation
  - User_ID             - Find all sessions for a user/employee
  - User_Type           - Filter by user type
  - Is_Active           - Filter active sessions
  - Expires_At          - Cleanup expired sessions

# ───────────────────────────────────────────────────────────────────────────────
#  TABLE: Session_Audit_Log
#  Purpose: Security audit trail for session-related events
# ───────────────────────────────────────────────────────────────────────────────

TABLE_NAME: Session_Audit_Log

COLUMNS:
┌─────────────────────┬──────────────┬──────────┬─────────────────────────────────────────┐
│ Column Name         │ Data Type    │ Required │ Description                             │
├─────────────────────┼──────────────┼──────────┼─────────────────────────────────────────┤
│ ROWID               │ bigint       │ Auto     │ Primary key (auto-generated by Catalyst)│
│ Event_Type          │ varchar      │ Yes      │ Type of event (see list below)          │
│ Session_ID          │ foreign key  │ No       │ FK to Sessions.ROWID (optional)         │
│ User_ID             │ foreign key  │ No       │ FK to Users/Employees.ROWID (optional)  │
│ User_Email          │ text         │ No       │ User's email at time of event           │
│ IP_Address          │ varchar      │ No       │ Client IP address                       │
│ Event_Timestamp     │ text         │ No       │ ISO 8601 timestamp of event             │
│ Details             │ text         │ No       │ JSON string with additional event data  │
│ Severity            │ varchar      │ No       │ INFO, WARNING, ERROR, CRITICAL          │
└─────────────────────┴──────────────┴──────────┴─────────────────────────────────────────┘

IMPORTANT NOTES:
  ⚠️  Session_ID and User_ID are FOREIGN KEYS:
      - Must reference valid ROWID from Sessions/Users/Employees tables
      - OR be completely omitted (NULL) - they are NOT mandatory
      - CANNOT be empty strings, zero, or negative integers
      - Code automatically omits these fields if no valid reference exists
  
  📝  User_ID can reference either:
      - Users.ROWID for passenger actions
      - Employees.ROWID for staff actions
  
  🔍  Session_ID stored in Details as "session_ref" since FK requires ROWID:
      - When actual Sessions.ROWID is unknown, session ID stored in Details JSON
      - Allows tracking session events even when session record not yet created

EVENT_TYPES:
  - SESSION_CREATED       : New session created on login/register
  - SESSION_VALIDATED     : Session successfully validated
  - SESSION_EXPIRED       : Session expired (timeout)
  - SESSION_REVOKED       : Session manually revoked by user
  - SESSION_REVOKED_ADMIN : Session revoked by administrator
  - SESSION_LIMIT_ENFORCED: Oldest session revoked due to concurrent limit
  - CSRF_VALIDATION_FAILED: CSRF token mismatch detected
  - SESSION_INVALID       : Invalid session ID presented
  - PASSWORD_CHANGED      : User changed password (all sessions revoked)
  - SUSPICIOUS_ACTIVITY   : Potential security threat detected

INDEXES (Create in Catalyst Console):
  - User_ID             - Find all events for a user
  - Event_Type          - Filter by event type
  - Event_Timestamp     - Time-based queries

EXAMPLE RECORD:
{
    "ROWID": "111222333444555",
    "Event_Type": "SESSION_CREATED",
    "Session_ID": "dGhpc2lz...xyz",

    "Event_Timestamp": "2026-03-31T12:00:00.000Z",

    "Severity": "INFO"
}
